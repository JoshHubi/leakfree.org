Use this script:

alias +sprint "+speed; unbind mouse1" alias -sprint "-speed; bind mouse1 +attack" bind shift +sprint

Or this script (altough untested):

alias +sprint "+speed; bind w +forward2; bind s +back2; bind a +moveleft2; bind d +moveright2"
alias -sprint "-speed; bind w +forward; bind s +back; bind a +moveleft; bind d +moveright"
alias +forward2 "+forward; unbind mouse1; unbind mouse2"
alias -forward2 "-forward; bind mouse1 +attack; bind mouse2 +attack2"
alias +back2 "+back; unbind mouse1; unbind mouse2"
alias -back2 "-back; bind mouse1 +attack; bind mouse2 +attack2"
alias +moveleft2 "+moveleft; unbind mouse1; unbind mouse2"
alias -moveleft2 "-moveleft; bind mouse1 +attack; bind mouse2 +attack2"
alias +moveright2 "+moveright; unbind mouse1; unbind mouse2"
alias -moveright2 "-moveright; bind mouse1 +attack; bind mouse2 +attack2"
bind shift +sprint
